
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() {
      return request.auth != null;
    }
    function isOwner(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }
    function isModerator() {
      let role = get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role;
      return isSignedIn() && (role == 'moderator' || role == 'admin');
    }

    match /users/{userId} {
      allow read: if isSignedIn();
      allow create: if isOwner(userId);
      allow update: if isOwner(userId) || (
        isSignedIn() && 
        request.resource.data.diff(resource.data).affectedKeys().hasOnly([
          'followersCount', 'followingCount', 'postsCount', 'likesReceived', 'viewsReceived', 'isOnline', 'lastActive'
        ])
      );
    }

    match /posts/{postId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn() && request.resource.data.authorId == request.auth.uid;
      allow update: if isSignedIn() && (
        isOwner(resource.data.authorId) || 
        isModerator() ||
        request.resource.data.diff(resource.data).affectedKeys().hasOnly([
          'likesCount', 'commentsCount', 'viewsCount', 'content', 'mediaURL', 'type'
        ])
      );
      allow delete: if isSignedIn() && (isOwner(resource.data.authorId) || isModerator());
    }

    match /blocks/{blockId} {
      allow read, delete: if isSignedIn() && resource.data.blockerId == request.auth.uid;
      allow create: if isSignedIn() && request.resource.data.blockerId == request.auth.uid;
    }

    match /saved_posts/{saveId} {
      allow read, create, delete: if isSignedIn() && (request.resource.data.userId == request.auth.uid || resource.data.userId == request.auth.uid);
    }

    match /comments/{commentId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn() && request.resource.data.authorId == request.auth.uid;
      allow update: if isSignedIn() && (
        isOwner(resource.data.authorId) || 
        isModerator() ||
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['likesCount', 'content', 'moderationStatus', 'replyCount', 'updatedAt'])
      );
      allow delete: if isSignedIn() && (isOwner(resource.data.authorId) || isModerator());
    }

    match /likes/{likeId} {
      allow read: if isSignedIn();
      allow create, delete: if isSignedIn() && (request.resource.data.userId == request.auth.uid || resource.data.userId == request.auth.uid);
    }

    match /comment_likes/{likeId} {
      allow read: if isSignedIn();
      allow create, delete: if isSignedIn() && (request.resource.data.userId == request.auth.uid || resource.data.userId == request.auth.uid);
    }
    
    match /follows/{followId} {
      allow read: if isSignedIn();
      allow create, delete: if isSignedIn() && (request.resource.data.followerId == request.auth.uid || resource.data.followerId == request.auth.uid);
    }

    match /notifications/{notifId} {
      allow read, update: if isSignedIn() && resource.data.recipientId == request.auth.uid;
      allow create: if isSignedIn();
    }

    match /chats/{chatId} {
      allow read, update: if isSignedIn() && request.auth.uid in resource.data.participants;
      allow create: if isSignedIn() && request.auth.uid in request.resource.data.participants;
    }

    match /messages/{messageId} {
      allow read: if isSignedIn(); 
      allow create: if isSignedIn() && request.resource.data.senderId == request.auth.uid;
      allow update: if isSignedIn() && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['status']);
    }
  }
}
